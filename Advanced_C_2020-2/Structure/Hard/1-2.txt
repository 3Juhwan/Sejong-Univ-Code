#pragma warning (disable:4996)
#include <stdio.h>
#include <string.h>

struct Student {
	char name[10];
	char number[11];
	char grade;
};

int main() {
	struct Student student[5];
	int i;
	char find[10] = "";
	char tmp_number[11] = "";
	char tmp_grade;
	int found = 0;

	for (i = 0; i < 5; i++)
	{
		scanf("%s %s %c", student[i].name, student[i].number, &student[i].grade);
		getchar();
	}

	gets_s(find, 10);

	for (i = 0;i < 5;i++)
	{
		if (strcmp(student[i].name, find) == 0)
		{
			strcpy(tmp_number, student[i].number);
			tmp_grade = student[i].grade;
			break;
		}
	}

	for (i = 0; i < 5; i++)
	{
		if (strcmp(student[i].name, find) == 0)
		{
			if (tmp_grade == student[i].grade && tmp_number[3] == student[i].number[3])
			{
				found = 1;
				printf(" %s", student[i].name);
			}
		}
	}
	if (found == 0)
		printf("%d\n", found);

	return 0;
}
