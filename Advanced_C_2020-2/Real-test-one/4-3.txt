#pragma warning (disable:4996)
#include <stdio.h>
#include <string.h>

struct person {
	char name[11];
	char birth[11];
	int score1;
	int score2;
	double avg;
};

void swap(struct person* p, struct person* q);

int main() {
	struct person person[100] = { '\0' }, * p = NULL, * q = NULL, tmp = { '\0' };
	int N = 0, M = 0;
	double max_score = 0;

	scanf("%d %d", &N, &M);
	getchar();

	for (p = person; p < person + N; p++)
	{
		scanf("%s %s %d %d", p->name, p->birth, &p->score1, &p->score2);
		getchar();

		p->avg = p->score1 * 0.8 + p->score2 * 0.2;
	}

	for (p = person; p < person + N - 1; p++)
	{
		for (q = p + 1; q < person + N; q++)
		{
			if (p->avg < q->avg)
			{
				swap(p, q);
			}

			else if (p->avg == q->avg)
			{
				if (p->score1 < q->score1)
				{
					swap(p, q);
				}

				else if (p->score1 == q->score1)
				{
					if (strcmp(p->birth, q->birth) < 0)
					{
						swap(p, q);
					}
				}
			}
		}
	}

	for (p = person; p < person + M; p++)
		printf("%s %.1lf\n", p->name, p->avg);

	return 0;
}

void swap(struct person* p, struct person* q) {
	struct person tmp;

	tmp = *p;
	*p = *q;
	*q = tmp;
}
