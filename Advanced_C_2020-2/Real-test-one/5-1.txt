#pragma warning (disable:4996)
#include <stdio.h>
#include <string.h>

struct word {
	char letters[21];
	int count;
};

void count_word(struct word* word, int N);

int main() {
	struct word W[100] = { '\0' }, * p = NULL, * q = NULL, tmp;
	int N = 0;

	scanf("%d", &N);
	getchar();

	for (p = W; p < W + N; p++)
		gets_s(p->letters, 21);

	count_word(W, N);

	for (int i = 0; i < N - 1; i++)
	{
		for (int j = 0; j < N; j++)
		{
			if (W[j].count < W[j + 1].count)
			{
				tmp = W[j];
				W[j] = W[j + 1];
				W[j + 1] = tmp;
			}
		}
	}

	for (p = W; p < W + N; p++)
		printf("%d %s\n", p->count, p->letters);

	return 0;
}

void count_word(struct word* word, int N) {
	struct word* p = word;
	int cnt = 0;

	for (p = word; p < word + N; p++)
	{
		cnt = 0;

		for (int i = 0; p->letters[i]; i++)
		{
			if (p->letters[i] == 'a' || p->letters[i] == 'e' || p->letters[i] == 'i' || p->letters[i] == 'o' || p->letters[i] == 'u' || p->letters[i] == 'A' || p->letters[i] == 'E' || p->letters[i] == 'I' || p->letters[i] == 'O' || p->letters[i] == 'U')
				cnt++;
		}
		p->count = cnt;
	}
}
