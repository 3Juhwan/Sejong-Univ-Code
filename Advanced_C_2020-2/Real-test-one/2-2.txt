#pragma warning (disable:4996)
#include <stdio.h>
#include <string.h>

int main() {
	char arr[101] = "", Larr[101] = "", Sarr[101] = "";
	char* Lword = NULL, * Sword = NULL;
	char* p = NULL, * q = NULL;
	int N = 0, min = 101, max = -1;
	int len = 0;

	scanf("%d", &N);
	getchar();

	for (int i = 0; i < N; i++)
	{
		gets_s(arr, 101);
		len = strlen(arr);

		if (len < min) {
			min = len;
			strcpy(Sarr, arr);
		}

		if (len > max) {
			max = len;
			strcpy(Larr, arr);
		}
	}

	min = 101;
	p = Sarr;
	q = Sarr;

	while (*p)
	{
		if (*p == ' ')
		{
			*p = '\0';
			len = strlen(q);

			if (len < min) {
				min = len;
				Sword = q;
			}
			q += strlen(q) + 1;
		}
		p++;
	}
	printf("%s\n", Sword);

	max = -1;
	p = Larr;
	q = Larr;

	while (*p)
	{
		if (*p == ' ')
		{
			*p = '\0';
			len = strlen(q);

			if (len > max) {
				max = len;
				Lword = q;
			}
			q += strlen(q) + 1;
		}
		p++;
	}

	printf("%s", Lword);

	return 0;
}
